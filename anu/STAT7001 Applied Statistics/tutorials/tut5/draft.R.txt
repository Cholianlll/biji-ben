crab <- read.csv("crab.csv", header=T)
head(crab)
attach(crab)
levels(SPECIES)

SPECIES <- as.factor(SPECIES)

crab.lm <- lm(log(FORCE) ~ log(HEIGHT)+SPECIES)

t.test(log(crab[crab$SPECIES!='Lophopanopeus bellus',]$FORCE) ~ crab[crab$SPECIES!='Lophopanopeus bellus',]$SPECIES, data=crab, conf.level=.95) # t-test

anova(crab.lm) # yes

summary(crab.lm) # nope
