#QUESTION 4

X=1:100
n=length(X)
Y=rep(0,n)
numsamp=1000
MeanResponse=rep(0,numsamp)
Xnew=2.5
set.seed(1)
for(i in 1:numsamp){
  errors=rnorm(n, 0, 1)
  Y=1+0*X+errors
  SLRfit=lm(Y~X)
  MeanResponse[i]=SLRfit$coef[1]+SLRfit$coef[2]*Xnew
}
hist(MeanResponse)
mean(MeanResponse)
sd(MeanResponse)
